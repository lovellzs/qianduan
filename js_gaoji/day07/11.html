示例代码

//01 提取成公共的发行者对象
    var publisher = {
        users:{
            default:[]
        },   //所有的订阅者对象
        addUser:function (fn,type) {
            //对订阅的类型进行判断
            var type = type || "default";
            //如果是新的订阅类型,那么我们需要做初始化的处理
            if (this.users[type] == undefined)
            {
                this.users[type] = [];
            }

            this.users[type].push(fn);
        },
        removeUser:function (fn,type) {
            this.tool(type,"removerUser",fn);
        },
        publish:function (type) {
            this.tool(type,"publish");
        },
        tool:function (type,funcType,fn) {
            var type = type || "default";
            var users = this.users[type];
            if (users == undefined)
            {
                console.log("当前没有任何订阅者");
                return;
            }
            for (var i = 0; i < users.length; i++) {
                if (funcType == 'publish')
                {
                    users[i]()
                }else
                {
                    if (users[i] == fn)
                    {
                        users.splice(i, 1);
                        break;
                    }
                }

            }
        }
    };

    //02 提供一个工具函数,能够利用发行者对象的模板来快速创建新的发布者
    function makePaper(o) {
        for(var i in publisher) {
            if (publisher.hasOwnProperty(i) && typeof publisher[i] == 'function')
            {
                o[i] = publisher[i];
            }
        }
        o.users = {default:[]};
    }

    //03 创建发布者
    var paper1 = {
        day:function () {
            this.publish();
        },
        month:function () {
            this.publish("month")
        },
        year:function () {
            this.publish("year")
        }
    };

    //把某个对象编程一个发布者
    makePaper(paper1);

    //04 创建订阅者(当发布者发布消息的时候,订阅者能够收到信息--自动调用订阅者的方法)
    var obj = {
        lookDayNews:function () {
            console.log("obj - 查看最新的日报信息");
        },
        lookImageNews:function () {
            console.log("obj - 查看图片信息");
        }
    };

    //05 设置订阅
    paper1.addUser(obj.lookDayNews);
    paper1.addUser(obj.lookImageNews,'month');
    paper1.day();
    paper1.month();
</script>
