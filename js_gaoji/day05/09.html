第五天笔记

变量和函数提升

JS中的代码执行分为两个步骤

① 预解析

    JavaScript在预解析阶段,会对使用var关键字声明的变量和function声明的代码块进行提升操作,提升到当前作用域的顶端

② 执行

代码提升的几种情况

01 函数提升

<script>

    func();

    function func() {
        console.log("测试的函数");
    }
</script>

02 变量提升

<script>
    console.log(a);     //打印出来的结果为undefined
    var a = 10;

    //var a;                //注意:只会对变量的声明进行提升
    //console.log(a);
    //a = 10
</script>

03 函数同名情况提升

<script>
    func01();                   //打印last
    function func01() {
        console.log("first");
    }

    func01();                   //打印last
    function func01() {
        console.log("last");
    }

    //模拟提升后的情况
    function func01() {
        console.log("first");
    }

    function func01() {
        console.log("last");
    }

    func01();
    func01();

</script>

说明：预处理的时候，同名的函数都会进行提升，但是后面的会覆盖掉前面的

04 变量名和函数同名的情况

<script>

    console.log(a);                     //打印function
    function a() {
        console.log("我是一个函数");
    }
    var a = 20;
    console.log(a);                     //打印20


    //变量和函数提升后的结果 错误
//    function a() {
//        console.log("我是一个函数");
//    }
//    var a ;
//    console.log(a);
//    a = 20;
//    console.log(a);

    //变量和函数提升后的结果 正确
    function a() {
        console.log("我是一个函数");
    }
    console.log(a);
    var a = 20;
    console.log(a);

</script>

总结:如果出现变量和函数同名的情况,则在进行提升的时候,只会提升函数到当前作用域顶端而忽略变量的提升操作
